case = int(input())

for i in range(case):
    k = int(input())
    n = int(input())
    a_0 = list(range(1, n+1)) #0ì¸µ
    matrix = [a_0]
    for j in range(1, k+1):
        matrix.append([])
    for l in range(1, k+1):
        for p in range(n):
            matrix[l].append(sum(matrix[l-1][:p+1]))
    print(matrix[k][n-1])
